This code was taken from my JS script for my portal page. I altered the original function to create a named function with toggleMode();, function expression with the variable toggleMode, then a arrow function. The function uses an event listener on a button, when a user clicks the button in will change the styling of the page to either light or dark.

Named Function

            function toggleMode() {
                if (mode === 'dark') {
                    body.className = 'switch';
                    banner.className = 'switch';
                    button.className = 'switch';
                    header.className = 'switch';
                    for (const section of sections) {
                        section.className = 'switch';
                    }
                    mode = 'light';
                } else {
                    body.removeAttribute('class');
                    banner.removeAttribute('class');
                    button.removeAttribute('class');
                    header.removeAttribute('class');
                    for (const section of sections) {
                        section.removeAttribute('class');
                    }
                    mode = 'dark';
                }
            }
            
            button.addEventListener('click', toggleMode);
